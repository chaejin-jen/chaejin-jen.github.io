    sequelize란

    Sequelize 는 ORM(Object-Relational Mapping)

    • Nodejs로 DB(mysql 또는 postgresql)를 제어할 수 있게 해줌
    • ORM : 객체와 관계형 데이터베이스(RBDMS) 관계를 매핑해주는 도구
      • Query문을 사용하지 않고 CRUD를 가능하게 해줌

    비유하자면.. 웹용 엑셀!

    sequelize-auto

    sequelize를 사용할 때 각 데이터베이스의 스키마 정의가 필요

    sequelize-auto : 스키마 정의를 사람 손을 거치지 않고 자동으로 생성해주는 편리한 도구

    • 특히, 이미 존재하는 데이터베이스의 스키마의 작업에 용이
    • express 서버 시작 시에 sequelize-auto를 사용하여 스키마 파일을 생성하게 하는 방법 추천

    dotenv 설정

    Node.js로 개발을 진행시 환경변수들을 관리하기 편하기 위한 설정
    (환경변수 설정)

    local pc 에서 개발 진행시 전역으로 필요한 여러 정보들이 존재
    (ID, PW , Access token, Facebook client ID /Secret ID … 등)
    얘네를 하나하나 index.js 로 가서 입력하고 바꿔주고 하기 번거로움

    $\rightarrow$ .env 파일에 설정해 놓게 됨

    주로 활용하는 옵션

    sequelize-auto -o "./models" -d databasename -h host -u username -p port -x password -t tablename -e rdbmstype

    옵션 해석
    -o: 스키마 파일을 생성할 경로
    -d: 데이터베이스(혹은 스키마) 이름
    -h: 데이터베이스 host
    -p: 데이터베이스 port
    -u: 접속 아이디
    -x: 패스워드
    -t: 특정 테이블만 가져오고 싶은 경우 추가 (없으면 전체 테이블을 생성함)
    -e: 데이터베이스 타입 (mysql, postgre 등)

    host: process.env.DB_HOST,
    dialect: 'mysql',
    timezone: '+09:00', //한국 시간 셋팅
    operatorsAliases: Sequelize.Op,
    pool: {
        max: 5,
        min: 0,
        idle: 10000
    }

let db = [];

// sync 명령 (index.js 를 제외한 각각 프로젝트의 .js 참조하여(=sync를 걸어서) 테이블을 생성하는 코드)
fs.readdirSync(__dirname)
    .filter(file => {
        return file.indexOf('.js')&& file !== 'index.js'
    })
    .forEach(file => {
        var model = sequelize.import(path.join(__dirname,
            file));
            db[model.name] = model;
    });

Object.keys(db).forEach(modelName => {
    if("associate" in db[modelName]){
        db[modelName].associate(db); // 모델간의 관계 (모델 작성에서 설명)
    }
});

db.sequelize = sequelize;

    app.js 
    const express = require('express');
const nunjucks = require('nunjucks');
const logger = require('morgan');

// db 관련
const db = require('./models');

class App {

    constructor () {
        this.app = express();
        
        // 뷰엔진 셋팅
        this.setViewEngine();

        // db 접속
        this.dbConnection();

        // 미들웨어 셋팅
        this.setMiddleWare();

        // 정적 디렉토리 추가
        this.setStatic();

        // 로컬 변수
        this.setLocals();

        // 라우팅
        this.getRouting();

        // 404 페이지를 찾을수가 없음
        this.status404();

        // 에러처리
        this.errorHandler();

    }
    
    dbConnection(){
        // DB authentication
        db.sequelize.authenticate()
        .then(() => {
            console.log('Connection has been established successfully.');
        })
        .then(() => {
            console.log('DB Sync complete.');
        })
        .catch(err => {
            console.error('Unable to connect to the database:', err);
        });
    }

    setMiddleWare (){
        
        // 미들웨어 셋팅
        this.app.use(logger('dev'));
        this.app.use(express.urlencoded({ extended: false }));
        this.app.use(express.json());

    }

    setViewEngine (){

        process.env.DB_USER

        nunjucks.configure('template', {
            autoescape: true,
            express: this.app
        });

    }

    setStatic (){
        this.app.use('/uploads', express.static('uploads'));
    }

    setLocals(){

        // 템플릿 변수
        this.app.use( (req, res, next) => {
            this.app.locals.isLogin = true;
            this.app.locals.req_path = req.path;
            next();
        });

    }

    getRouting (){
        this.app.use(require('./controllers'))
    }

    status404() {        
        this.app.use( ( req , res, _ ) => {
            res.status(404).render('common/404.html')
        });
    }

    errorHandler() {

        this.app.use( (err, req, res,  _ ) => {
            res.status(500).render('common/500.html')
        });
    
    }

}

module.exports = new App().app;

module.exports = function(sequelize, DataTypes){
    const Products = sequelize.define('Products',
        {
            id: { type: DataTypes.INTEGER, primaryKey: true, autoIncrement: true },
            name : { type: DataTypes.STRING },
            price : { type: DataTypes.INTEGER },
            description : { type: DataTypes.TEXT }
        }
    );

        moment(date).format('YYYY-MM-DD')
    );
    
    return Products;
}

      • Models.테이블명.create(데이터) : sequelize 테이블에 1row를 입력
        • bulkCreate : row여러개를 한꺼번에 insert
      • Models.테이블명.findAll(조회조건) : sequelize에서 여러줄을 조회
      • Models.테이블명.findByPk(primart key)
      • Models.테이블명.findOne(조회조건)
      • Models.테이블명.update(데이터, 조회조건)
      • Models.테이블명.destroy(조회조건)
